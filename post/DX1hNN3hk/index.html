数组中第k个最大的元素

找数组中第k大的元素

class Solution {

    /**
     * 将其由小到大排序，选定一个位置作为分割
     * 找第k大的数
     * @param nums
     * @param k
     * @return
     */
    public int findKthLargest(int[] nums, int k) {
        // 首先使用倒数k作为分割点进行查找
        return findK(nums, 0, nums.length -1, nums.length - k);
    }

    /**
     *
     * @param nums 数组
     * @param start 开始
     * @param end 结束
     * @param k 第k大
     * @return
     */
    public int findK(int[] nums, int start, int end, int k) {
        if (start == end) {
            return nums[start];
        }
        int pivot = nums[(start + end)/2];
        int i = start;
        int j = end;
        while (i <= j) {
            while (nums[i] < pivot) {
                i++;
            }
            while (nums[j] > pivot) {
                j--;
            }
            if (i <= j) {
                int temp = nums[i];
                nums[i] = nums[j];
                nums[j] = temp;
                i++;
                j--;
            }
        }
        // 判断k在选定点前面还是后面，对选定的部分进行排序
        if (k >= (i-start)) {
            return findK(nums, i, end, k-i+start);
        } else {
            return findK(nums, start, i-1, k);
        }
    }
}

最长连续序列

class Solution {
    /**
     * 最长连续序列，注意最长连续序列中的任意一个数
     * @param nums
     * @return
     */
    public int longestConsecutive(int[] nums) {
        // 记录该字符所在的最长子序列的序列长度
        HashMap<Integer, Integer> map = new HashMap<>();
        // 最长子序列长度
        int res = 0;
        for (int t : nums) {
            // 如果map中包含说明已经计算过
            if (map.containsKey(t)) {
                continue;
            }
            // 获取t-1所在的最长连续序列的长度
            int left = map.getOrDefault(t - 1, 0);
            // 获取t+1所在的最长连续序列的长度
            int right = map.getOrDefault(t + 1, 0);
            // 最终长度
            int count = left + right + 1;
            // 更新t、t-1、t+1的值
            map.put(t, count);
            if (left > 0) {
                map.put(t - left, count);
            }
            if (right > 0) {
                map.put(t + right, count);
            }
            // 更新最长长度
            res = Math.max(res, count);
        }
        return res;
    }
}

螺旋矩阵

/**
 * 顺时针螺旋顺序
 */
class Solution {
    public List<Integer> spiralOrder(int[][] matrix) {
        List<Integer> ans = new ArrayList<>();
        if (matrix.length == 0) {
            return ans;
        }

        int startR = 0, endR = matrix.length - 1;
        int startC = 0, endC = matrix[0].length - 1;

        while (startR <= endR && startC <= endC) {
            // 上面一行遍历
            for (int c = startC; c <= endC; c++) {
                ans.add(matrix[startR][c]);
            }
            // 右边一列遍历，起始位置应该start+1
            for (int r = startR + 1; r <= endR; r++) {
                ans.add(matrix[r][endC]);
            }
            // 倒序遍历
            if (startR < endR && startC < endC) {
                // 下面一行遍历:从倒数第二个到第一个
                for (int c = endC - 1; c > startC; c--) {
                    ans.add(matrix[endR][c]);
                }
                // 左边一行遍历:从倒数第二个到正数第二个
                for (int r = endR; r > startR; r--) {
                    ans.add(matrix[r][startC]);
                }
            }
            // 矩阵内缩
            startR++;
            endR--;
            startC++;
            endC--;
        }
        return ans;
    }
}

螺旋矩阵II

class Solution {
    public int[][] generateMatrix(int n) {
        int[][] res = new int[n][n];
        int startC = 0, endC = n - 1;
        int startR = 0, endR = n - 1;
        int i = 1;
        while (startC <= endC && startR <= endR) {
            for (int c = startC; c <= endC; c++) {
                res[startR][c] = i++;
            }
            for (int r = startR + 1; r <= endR; r++) {
                res[r][endC] = i++;
            }
            if (startC < endC && startR < endR) {
                for (int c = endC - 1; c > startC; c--) {
                    res[endR][c] = i++;
                }
                for (int r = endR; r > startR; r--) {
                    res[r][startC] = i++;
                }
            }
            startC++;
            startR++;
            endC--;
            endR--;
        }
        return res;
    }
}

买卖股票的最佳时机II

class Solution {
    public int maxProfit(int[] prices) {
        if (prices.length < 2) {
            return 0;
        }
        int max = prices[prices.length - 1];
        int min = max;
        int res = 0;
        for (int i = prices.length - 1; i >= 0; i--) {
            if (prices[i] >= max) {
                res = res + max - min;
                max = prices[i];
                min = max;
            } else if (prices[i] <= min) {
                min = prices[i];
            } else {
                res = res + max - min;
                max = prices[i];
                min = max;
            }
        }
        res = res + max - min;
        return res;
    }
}

合并两个有序数组

给出两个有序数组，将B数组合并进A数组
方法：倒序插入
获得合并后的尾指针和两个数组的尾指针，逐渐向内移动尾指针，比较两个数组中的值，决定将哪一个赋值给对应的位置

class Solution {
    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int i = m + n - 1, j = m - 1, k = n - 1;
        while (j >= 0 && k >= 0) {
            nums1[i--] = (nums1[j] > nums2[k]) ? nums1[j--] : nums2[k--];
        }
        while (k >= 0) {
            nums1[i--] = nums2[k--];
        }
    }
}

被围绕的区域

  • DFS
class Solution {
    public void solve(char[][] board) {
        if(board == null || board.length == 0) return;
        int m = board.length;
        int n = board[0].length;
        // 对于在边界出现的O进行DFS
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                boolean isEdge = (i == 0 || j == 0 || i == m - 1 || j == n - 1);
                if (isEdge && board[i][j] == 'O') {
                    dfs(board, i, j);
                }
            }
        }
        // DFS后，无法替换的O已经被置换成#，这里将其置换回来
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (board[i][j] == 'O') {
                    board[i][j] = 'X';
                }
                if (board[i][j] == '#') {
                    board[i][j] = 'O';
                }
            }
        }
    }

    /**
     * 深度优先遍历递归
     * @param board
     * @param i
     * @param j
     */
    public void dfs(char[][] board, int i, int j) {
        // 对非边界的O进行查找，替换为#
        if (i < 0 || j < 0 || i >= board.length || j >= board[0].length || board[i][j] == 'X' || board[i][j] == '#') {
            return;
        }
        board[i][j] = '#';
        dfs(board, i - 1, j);
        dfs(board, i + 1, j);
        dfs(board, i, j - 1);
        dfs(board, i, j + 1);
    }

}

分割回文串

class Solution {
    private String s;
    private List<List<String>> result = new ArrayList<>();

    /**
     * 验证s[i,j]子串是否是回文
     * @param i i
     * @param j j
     * @return a[i,j]是否为回文
     */
    private boolean validate(int i, int j){
        while(i < j){
            if(s.charAt(i) != s.charAt(j)) {
                return false;
            }
            i++;
            j--;
        }
        return true;
    }

    /**
     *
     * @param s 字符串
     * @param list  将分割的字符串存储在 list中
     * @param index index，在字符串中的下标
     */
    private void f(String s, List<String> list, int index){
        // 当下标移动至结尾时，就可以将其添加到result中
        if(index >= s.length()){
            result.add(new ArrayList<>(list));
            return;
        }
        // 从index开始判断s[index, i]子串是否是回文串
        for(int i = index; i < s.length(); i++){
            // 如果是回文串，放入list中，然后对
            if(validate(index, i)){
                list.add(s.substring(index, i + 1));
                f(s, list, i + 1);
                list.remove(list.size() - 1);
            }
        }
    }

    public List<List<String>> partition(String s) {
        if(s.equals("")||s.length()==0) {
            return result;
        }
        this.s=s;
        f(s, new ArrayList<>(), 0);
        return result;
    }
}
